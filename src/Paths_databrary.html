module Paths_databrary (
    version,
    getBinDir, getLibDir, getDataDir, getLibexecDir,
    getDataFileName, getSysconfDir
  ) where

import qualified Control.Exception as Exception
import Data.Version (Version(..))
import System.Environment (getEnv)
import Prelude

catchIO :: IO a -> (Exception.IOException -> IO a) -> IO a
catchIO = Exception.catch

version :: Version
version = Version [1,3,1,2] ["gf874494","dirty"]
bindir, libdir, datadir, libexecdir, sysconfdir :: FilePath

bindir     = "/home/dylan/.cabal/bin"
libdir     = "/home/dylan/.cabal/lib/x86_64-linux-ghc-7.10.3/databrary-1.3.1.2-D3J75TxWypH8dw0ZYN9iCu"
datadir    = "/home/dylan/.cabal/share/x86_64-linux-ghc-7.10.3/databrary-1.3.1.2"
libexecdir = "/home/dylan/.cabal/libexec"
sysconfdir = "/home/dylan/.cabal/etc"

getBinDir, getLibDir, getDataDir, getLibexecDir, getSysconfDir :: IO FilePath
getBinDir = catchIO (getEnv "databrary_bindir") (\_ -> return bindir)
getLibDir = catchIO (getEnv "databrary_libdir") (\_ -> return libdir)
getDataDir = catchIO (getEnv "databrary_datadir") (\_ -> return datadir)
getLibexecDir = catchIO (getEnv "databrary_libexecdir") (\_ -> return libexecdir)
getSysconfDir = catchIO (getEnv "databrary_sysconfdir") (\_ -> return sysconfdir)

getDataFileName :: FilePath -> IO FilePath
getDataFileName name = do
  dir <- getDataDir
  return (dir ++ "/" ++ name)
